\entrybig
    {\textbf{General Loop Packing Analysis and Transformation}}{}
    {Graduate Research Collaboration}{2021\textendash2022}
\innerlist{
    \entry{Packing is an important optimization used in
            high-performance computing domains to improve data
            locality of tensor accesses within loop nests.
            In this work, we present a profitability analysis
            to determine when and where packing could be beneficial as
            a transformation on a general computation. We provide an
            implementation as an MLIR compiler pass that performs
            the analysis and code transformation.}
}
